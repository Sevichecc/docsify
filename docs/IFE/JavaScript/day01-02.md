#### **Task 02**

```js
function calculateSum(price, vip) {
  price = vip ? price * 0.98 : price;

  if (price >= 300) {
    price -= 30;
  } else if (price >= 200) {
    price -= 10;
  } else if (price >= 100) {
    price -= 5;
  }
  return price;
}

console.log(calculateSum(300, false)); //->270
console.log(calculateSum(300, true)); //->284
```
